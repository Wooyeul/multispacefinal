#mypage_qna_find_user_info_proc
DROP PROCEDURE IF EXISTS p_mypage_qna_find_user_info;
DELIMITER $$
CREATE PROCEDURE p_mypage_qna_find_user_info(
	IN v_user_id VARCHAR(20)
)
BEGIN

END$$
